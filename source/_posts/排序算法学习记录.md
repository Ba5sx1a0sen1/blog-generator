### 冒泡排序算法（JavaScript）

```
    function bubbleSort(array){
        let i,j,temp //声明外层与内层循环计数器，与数值交换中间变量
        let len = array.length //记录数组长度
        for(i = 0; i < len - 1 ; i++){
            for(j = 0; j < len - 1 -i; j++){
                if(array[j]>array[j+1]){
                    t = array[j]
                    array[j] = array[j+1]
                    array[j+1] = array[j]
                }
            }
        }
    }
```
